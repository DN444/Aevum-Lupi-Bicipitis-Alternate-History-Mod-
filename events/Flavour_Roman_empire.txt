country_event = { #Iberian Romans
	id = roman_emp.100
	title = roman_emp.100.test
	desc = roman_emp.100.desc
	picture = COMET_SIGHTED_eventPicture
	hidden = yes
	trigger = {
		any_owned_province = {
			culture = romano_iberic
		}
		NOT = {
			iberia_region = {
				type = all
				culture = romano_iberic
			}
		}
	}
	mean_time_to_happen = {
		months = 60
		modifier = {
			factor = 0.5
			has_country_flag = faster_romanization_flag
		}
		modifier = {
			factor = 0.25
			has_institution = enlightenment
		}
		modifier = {
			factor = 0.75
			full_idea_group = humanist_ideas
		}
		modifier = {
			factor = 0.75
			full_idea_group = religious_ideas
		}
	}
	option = {
		name = roman_emp.100a
		random_owned_province = {
			limit = {
				region = iberia_region
				NOT = {
					culture = romano_iberic
				}
			}
			change_culture = romano_iberic
		}
		if = {
			limit = { has_country_flag = rom_double_romanization}
			random_owned_province = {
				limit = {
					region = iberia_region
					NOT = {
						culture = romano_iberic
					}
				}
				change_culture = romano_iberic
			}
			random_owned_province = {
				limit = {
					region = iberia_region
					NOT = {
						culture = romano_iberic
					}
				}
				change_culture = romano_iberic
			}
		}
		random = {
			chance = 50 
			random_owned_province = {
				limit = {
					region = iberia_region
					NOT = {
						culture = romano_iberic
					}
				}
				change_culture = romano_iberic
			}
			random_owned_province = {
				limit = {
					region = iberia_region
					NOT = {
						culture = romano_iberic
					}
				}
				change_culture = romano_iberic
			}
		}
	}
}

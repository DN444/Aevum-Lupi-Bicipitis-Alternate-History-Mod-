	}

	option = {
		name = frankia.2a
		if = {
			limit = {
				any_country = {
					ai = no
					culture_group = frankish
					NOT = {
						tag = VR0
					}
				}
			}
			every_country = {
				limit = {
					ai = no
					culture_group = frankish
					NOT = {
						tag = VR0
					}
				}
				country_event = {
					id = frankia.3
				}
			}
		}else = {
			if = {
				limit = {
					culture_group = frankish
				}
				change_tag = FR9
			}else = {
				every_country = {
					limit = {
						ai = yes
						culture_group = frankish
						NOT = {
							tag = VR0
						}
					}
					change_tag = FR9
				}
			}	
		}	
		FR9 = {
			add_country_modifier = {
				name = frankish_fury
				duration = 10950
			}
		}
	}
}

Ellan_1_slot = {
	slot = 1
	generic = no
	ai = yes
	has_country_shield = yes
	potential = {
		OR = {
			tag = FC8
			tag = FC3
			tag = FK0
			tag = FC4
			tag = FC6
			tag = FC7
			tag = FK1
			tag = FC9
			tag = FC5
		}
	}

	1_ellan_conf = {
		icon = king_of_france
		position = 1
		provinces_to_highlight = {
			OR = {
				is_capital_of = FC1
				is_capital_of = FC2
				is_capital_of = FC0
			}
		}

		trigger = {
			OR = {
				alliance_with = FC1
				alliance_with = FC2
				alliance_with = FC0
			}
		}

		effect = {
			every_known_country = {
				limit = { primary_culture = ellan } 
				reverse_add_casus_belli = {
					type = cb_vassalize_mission
					target = THIS
				}
			}
			every_province = {
				limit = { culture = ellan } 
				add_permanent_claim = ROOT
			}
		}
	}

	1_ellan_port = {
		icon = mission_invade_island
		position = 3
		provinces_to_highlight = {
			owned_by = ROOT
			has_port = yes
		}

		trigger = {
			any_owned_province = {
				has_port = yes
				OR = {
					has_building = dock
					has_building = drydock
				}
			}
			num_of_light_ship = 10
		}

		effect = {
			if = {
				limit = {
					NOT = {
						any_owned_province = {
							has_port = yes
							OR = {
								has_building = dock
								has_building = drydock
							}
						}
					}
				}
				custom_tooltip = little_custom_tip_1ep
			}
			else = {
				random_owned_province = {
					limit = {
						has_port = yes
						OR = {
							has_building = dock
							has_building = drydock
						}
					}
					if = {
						limit = { has_building = marketplace } 
						add_base_production = 2
					}
					else = { add_building = marketplace } 
				}
			}
		}
	}

	1_ellan_navy = {
		icon = mission_marines
		position = 4
		required_missions = { 1_ellan_port } 
		provinces_to_highlight = {  } 

		trigger = {
			navy_size_percentage = 0.85
			navy_tradition = 20
		}

		effect = { create_admiral = { tradition = 75 }  } 
	}

	1_ellan_navy_power = {
		icon = mission_establish_high_seas_navy
		position = 5
		required_missions = { 1_ellan_navy } 
		provinces_to_highlight = { owned_by = ROOT } 

		trigger = {
			navy_tradition = 50
			num_of_owned_provinces_with = {
				value = 10
				OR = {
					has_building = shipyard
					has_building = grand_shipyard
				}
			}
		}

		effect = {
			add_country_modifier = {
				name = "ports_of_ellan"
				duration = -1
			}
		}
	}
}

Roman_empire_2_slot = {

	has_country_shield = yes

	iberian_woes = {
		icon = mission_mediterranean_islands
		position = 2
		required_missions = { when_in_rome } 
		provinces_to_highlight = { region = iberia_region } 

		trigger = {
			iberia_region = {
				type = all
				owned_by = ROOT
				num_of_buildings_in_province = 2
				NOT = {
					local_autonomy = 1
				}
			}
		}

		effect = {
			if = {
				limit = {
					primary_culture = romano_iberic
				}
				add_stability_or_adm_power = yes
				iberia_region = {
					add_base_manpower = 1
				}
			}else = {
				add_accepted_culture = romano_iberic
				iberia_region = {
					change_culture = romano_iberic
				}
			}
		}
	}
	cartago_repaired_est = {

			num_of_owned_provinces_with = {
				value = 30

		}

		effect = {

				name = roman_cool_admin

			}
			custom_tooltip = ROM_TT_4

				random_owned_province = {
					add_province_modifier = {
						name = rom_extra_administration
						duration = -1
					}
				}	
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						has_building = state_house
					}
					add_province_modifier = {
						name = rom_extra_administration
						duration = -1
					}
				}
			}
		}
	}

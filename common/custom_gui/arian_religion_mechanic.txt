custom_button = {
	name = arian_pacifism_button
	potential = {
		religion = arian
	}
	trigger = {
		is_at_war = no
		OR = {
			NOT = {
				had_recent_war = 5
			}
			has_country_modifier = arian_pacifism
		}
	}
	effect = {
		if = {
			limit = {
				has_country_modifier = arian_pacifism
			}
			remove_country_modifier = arian_pacifism
		}else = {
			add_country_modifier = {
				name = arian_pacifism
				duration = -1
			}
		}
	}
	tooltip = arian_pacifism_tt
}

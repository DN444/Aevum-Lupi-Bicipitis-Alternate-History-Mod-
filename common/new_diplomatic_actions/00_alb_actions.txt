

valtea_event_action = {
	category = covert
	require_acceptance = no
	is_visible = {
		has_government_mechanic = valtea_intel_mechanic
    }

	is_allowed = {
		FROM = {
			has_spy_network_from = {
				who = ROOT
				value = 25
			}
			ROOT = {
				NOT = {
					tech_difference = 2
				}
			}
			if = {
				limit = {
					has_country_modifier = valtea_defense
				}
				custom_trigger_tooltip = {
					tooltip = NOT_DONE_VALTEA_SPY_ACTION_IN_COUNTRY
					NOT = {
						has_country_modifier = valtea_defense
					}
				}
			}
		}
    }
	on_accept = {
		ROOT = {
			add_spy_network_in = {
				who = FROM
				value = -25
			}
			country_event = {
				id = valt.1
			}
		}	
		FROM = {
			add_country_modifier = {
				name = valtea_defense
				duration =  7300
			}
		}
    }

	ai_will_do = {
        always = yes
    }
}

friendship_action = {
	category = influence
	require_acceptance = no
	is_visible = {
		ROOT = {
			has_country_flag = ellan_easy_expansion
		}
    }

	is_allowed = {
		is_at_war = no
		treasury = 100
		FROM = {
			primary_culture = ellan
			is_at_war = no
			has_opinion = {
				who = ROOT
				value = 190
			}
			alliance_with = ROOT
		}
    }
	on_accept = {
		ROOT = {
			add_treasury = -100
			if = {
				limit = {
					from = {
						ai = yes
					}
				}
				vassalize = FROM
			}
			else = {
				FROM = {
					every_owned_province = {
						add_core = ROOT
					}
				}
			}
		}
    }
	ai_will_do = {
        always = no
    }
}

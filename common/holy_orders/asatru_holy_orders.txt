

ast_adm_order = {
    icon = GFX_holy_order_santiago
    trigger = {
        has_country_flag = asatru_order_unlocked
        religion = norse_ragnaroked
    }
    localization = holy_order
    cost = 50
    cost_type = adm_power
    per_province_effect = {
        add_base_tax = 1
        if = {
            limit = {
                OR = {
                    has_building = temple
                    has_building = cathedral
                }
            }
            add_province_modifier = {
                name = adm_order_buff
                duration = -1
            }
        }
    }
    per_province_abandon_effect = {
        add_base_tax = -1
        if = {
            limit = {
                has_province_modifier = adm_order_buff
            }
            remove_province_modifier = adm_order_buff
        }
    }
    modifier = {
        local_missionary_strength = 0.015
    }
    ai_priority = {
        factor = 2
        modifier = {
            factor = 2.0
            NOT = { religion = ROOT }
        }
    }
    color = {
        100 5 50
    }
}

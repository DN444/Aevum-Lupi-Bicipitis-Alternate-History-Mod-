estate_knight_lodges_disaster = {

	potential = {
		NOT = {
			current_age = age_of_discovery
		}
		OR = {
			is_or_was_tag = {tag = TTS}
			is_or_was_tag = {tag = PD8}
		}
	}
	
	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_knight_lodges
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_knight_lodges
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_knight_lodges
					influence = 100
				}
			}
		}
	}
	
	progress = {	
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_knight_lodges
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_knight_lodges
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_knight_lodges
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_knight_lodges
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		always = yes
	}
	
	on_start = saxo.100
	
	on_start_effect = {
		if = {
			limit = {
				has_dlc = "The Cossacks"
			}
			custom_tooltip = estate_knight_lodges_disaster_tooltip_cossacks
		}
		else = {
			custom_tooltip = estate_knight_lodges_disaster_tooltip
		}
	}
}
